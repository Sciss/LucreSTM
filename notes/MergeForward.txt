trait Targets {
   def propagate( visited: mutable.Map[ S#ID, Int], parent: Event[ S, _ ], reactions: Reactions ) : Reactions = {
      children.foldLeft( reactions ) { case (rs, (key, child) =>

         if( (bitset & key) == 0 ) {
